# 基于本地知识图谱的知识库实现方案

## 一、基础概念科普

### 1.1 什么是知识图谱？

**知识图谱（Knowledge Graph, KG）** 是一种用图结构表示知识的技术，由**实体（Entity）**和**关系（Relation）**组成：

- **实体（节点）**：表示具体概念，如"光合作用"、"叶绿体"、"二氧化碳"
- **关系（边）**：表示实体间的关联，如"需要"、"产生"、"包含于"

> 类比理解：知识图谱就像一张巨大的概念地图，把零散的知识点连接成网络。

### 1.2 知识图谱 vs 传统数据库

| 特性 | 关系型数据库 | 知识图谱 |
|------|-------------|----------|
| **数据结构** | 表格（行和列） | 图（节点和边） |
| **查询方式** | SQL 语句 | 图遍历查询 |
| **擅长场景** | 精确匹配、统计计算 | 关系推理、路径发现 |
| **扩展性** |  schema 变更困难 | 灵活添加新关系 |

### 1.3 为什么教育场景需要知识图谱？

```
传统 RAG 检索：
用户问："光合作用需要什么条件？"
检索结果：包含"光合作用"关键词的文本片段
问题：可能遗漏"光照"、"叶绿体"等相关概念

知识图谱增强：
用户问："光合作用需要什么条件？"
图谱查询：找到"光合作用"节点 → 遍历"需要"关系边
结果：[光照, 二氧化碳, 水, 叶绿体]
优势：完整、结构化、可解释
```

### 4.1 Neo4j 知识图谱操作类

```python
from neo4j import GraphDatabase
from typing import List, Dict, Optional, Tuple
import json

class EducationKnowledgeGraph:
    """教育知识图谱管理类"""
    
    def __init__(self, uri: str, user: str, password: str):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))
        self._init_schema()
    
    def close(self):
        self.driver.close()
    
    def _init_schema(self):
        """初始化图数据库 Schema"""
        with self.driver.session() as session:
            # 创建约束和索引
            constraints = [
                "CREATE CONSTRAINT concept_name IF NOT EXISTS FOR (c:Concept) REQUIRE c.name IS UNIQUE",
                "CREATE CONSTRAINT chapter_id IF NOT EXISTS FOR (ch:Chapter) REQUIRE ch.id IS UNIQUE",
                "CREATE INDEX concept_subject IF NOT EXISTS FOR (c:Concept) ON (c.subject)",
                "CREATE INDEX concept_grade IF NOT EXISTS FOR (c:Concept) ON (c.grade)",
            ]
            for constraint in constraints:
                try:
                    session.run(constraint)
                except Exception as e:
                    print(f"Constraint creation warning: {e}")
    
    def create_concept(self, name: str, subject: str, grade: str, 
                       definition: str, properties: Dict = None) -> bool:
        """创建知识点实体"""
        query = """
        MERGE (c:Concept {name: $name})
        SET c.subject = $subject,
            c.grade = $grade,
            c.definition = $definition,
            c.created_at = datetime(),
            c.properties = $properties
        RETURN c
        """
        with self.driver.session() as session:
            result = session.run(query, name=name, subject=subject, 
                               grade=grade, definition=definition,
                               properties=json.dumps(properties or {}))
            return result.single() is not None
    
    def create_relationship(self, from_name: str, to_name: str, 
                           rel_type: str, properties: Dict = None) -> bool:
        """创建实体间关系"""
        query = f"""
        MATCH (a:Concept {{name: $from_name}})
        MATCH (b:Concept {{name: $to_name}})
        MERGE (a)-[r:{rel_type}]->(b)
        SET r.properties = $properties,
            r.created_at = datetime()
        RETURN r
        """
        with self.driver.session() as session:
            result = session.run(query, from_name=from_name, to_name=to_name,
                               properties=json.dumps(properties or {}))
            return result.single() is not None
    
    def get_concept(self, name: str) -> Optional[Dict]:
        """获取知识点详情"""
        query = """
        MATCH (c:Concept {name: $name})
        OPTIONAL MATCH (c)-[r]->(related:Concept)
        RETURN c, collect({relation: type(r), target: related.name}) as outgoing
        """
        with self.driver.session() as session:
            result = session.run(query, name=name)
            record = result.single()
            if record:
                concept = dict(record["c"])
                concept["outgoing_relations"] = record["outgoing"]
                return concept
            return None
    
    def find_prerequisites(self, concept_name: str, depth: int = 2) -> List[Dict]:
        """查找前置知识点（学习路径）"""
        query = """
        MATCH path = (prereq:Concept)-[:PREREQUISITE_OF*1..$depth]->(target:Concept {name: $name})
        RETURN prereq, length(path) as distance
        ORDER BY distance
        """
        with self.driver.session() as session:
            result = session.run(query, name=concept_name, depth=depth)
            return [{"name": record["prereq"]["name"], 
                    "distance": record["distance"]} 
                   for record in result]
    
    def find_related(self, concept_name: str, limit: int = 10) -> List[Dict]:
        """查找相关知识点"""
        query = """
        MATCH (c:Concept {name: $name})-[r:RELATED_TO|REQUIRES|PRODUCES]-(related:Concept)
        RETURN related, type(r) as relation_type, r.properties as props
        LIMIT $limit
        """
        with self.driver.session() as session:
            result = session.run(query, name=concept_name, limit=limit)
            return [{"name": record["related"]["name"],
                    "relation": record["relation_type"],
                    "properties": json.loads(record["props"] or "{}")}
                   for record in result]
    
    def get_learning_path(self, start_concept: str, end_concept: str) -> List[Dict]:
        """获取从起点到终点的学习路径"""
        query = """
        MATCH path = shortestPath(
            (start:Concept {name: $start})-[:PREREQUISITE_OF|RELATED_TO*]-(end:Concept {name: $end})
        )
        RETURN [node in nodes(path) | node.name] as path_nodes,
               [rel in relationships(path) | type(rel)] as path_rels
        """
        with self.driver.session() as session:
            result = session.run(query, start=start_concept, end=end_concept)
            record = result.single()
            if record:
                nodes = record["path_nodes"]
                rels = record["path_rels"]
                # 构建路径描述
                path = []
                for i, node in enumerate(nodes):
                    path.append({"concept": node})
                    if i < len(rels):
                        path.append({"relation": rels[i]})
                return path
            return []
    
    def search_concepts(self, keyword: str, subject: str = None, 
                       limit: int = 10) -> List[Dict]:
        """搜索知识点"""
        if subject:
            query = """
            MATCH (c:Concept)
            WHERE c.name CONTAINS $keyword AND c.subject = $subject
            RETURN c
            LIMIT $limit
            """
            params = {"keyword": keyword, "subject": subject, "limit": limit}
        else:
            query = """
            MATCH (c:Concept)
            WHERE c.name CONTAINS $keyword
            RETURN c
            LIMIT $limit
            """
            params = {"keyword": keyword, "limit": limit}
        
        with self.driver.session() as session:
            result = session.run(query, **params)
            return [dict(record["c"]) for record in result]
    
    def get_knowledge_subgraph(self, concept_name: str, depth: int = 2) -> Dict:
        """获取知识子图（用于可视化）"""
        query = """
        MATCH path = (center:Concept {name: $name})-[:RELATED_TO|REQUIRES|PRODUCES|PREREQUISITE_OF*1..$depth]-(related)
        WITH center, related, relationships(path) as rels
        RETURN {
            nodes: collect(DISTINCT {id: center.name, label: center.name, group: center.subject}) + 
                   collect(DISTINCT {id: related.name, label: related.name, group: related.subject}),
            edges: collect(DISTINCT {from: startNode(rel).name, to: endNode(rel).name, 
                          label: type(rel), arrows: "to"} for rel in rels)
        } as subgraph
        """
        with self.driver.session() as session:
            result = session.run(query, name=concept_name, depth=depth)
            record = result.single()
            if record:
                return record["subgraph"]
            return {"nodes": [], "edges": []}
```

### 4.2 GraphRAG 检索实现

```python
from typing import List, Dict
import numpy as np

class GraphRAGRetriever:
    """GraphRAG 检索器：结合知识图谱和向量检索"""
    
    def __init__(self, kg: EducationKnowledgeGraph, vector_store):
        self.kg = kg
        self.vector_store = vector_store
    
    async def retrieve(self, query: str, subject: str = None, 
                      top_k: int = 5) -> Dict:
        """
        GraphRAG 检索主函数
        
        流程：
        1. 向量检索获取语义相关文档
        2. 从文档中提取关键实体
        3. 在知识图谱中扩展相关实体
        4. 合并排序返回结果
        """
        results = {
            "vector_results": [],
            "graph_results": [],
            "combined": []
        }
        
        # Step 1: 向量检索
        vector_results = await self._vector_search(query, subject, top_k * 2)
        results["vector_results"] = vector_results
        
        # Step 2: 实体提取和图谱扩展
        entities = self._extract_entities_from_results(vector_results)
        graph_results = self._graph_expand(entities, top_k)
        results["graph_results"] = graph_results
        
        # Step 3: 混合排序
        combined = self._hybrid_rank(vector_results, graph_results, query)
        results["combined"] = combined[:top_k]
        
        return results
    
    async def _vector_search(self, query: str, subject: str = None, 
                            top_k: int = 10) -> List[Dict]:
        """向量语义检索"""
        # 生成查询向量
        query_embedding = self.vector_store.encode(query)
        
        # 向量检索
        results = self.vector_store.search(query_embedding, top_k=top_k)
        
        # 按学科过滤
        if subject:
            results = [r for r in results if r.get("subject") == subject]
        
        return results
    
    def _extract_entities_from_results(self, vector_results: List[Dict]) -> List[str]:
        """从向量检索结果中提取关键实体"""
        entities = set()
        
        for result in vector_results:
            content = result.get("content", "")
            # 使用简单的实体匹配（实际应用中可用 NER 模型）
            # 这里假设内容中包含已知的概念名称
            known_concepts = self.kg.search_concepts("", limit=1000)
            concept_names = [c["name"] for c in known_concepts]
            
            for concept in concept_names:
                if concept in content:
                    entities.add(concept)
        
        return list(entities)
    
    def _graph_expand(self, entities: List[str], top_k: int) -> List[Dict]:
        """基于知识图谱扩展相关实体"""
        expanded = []
        
        for entity in entities[:3]:  # 限制扩展的实体数量
            # 获取直接相关实体
            related = self.kg.find_related(entity, limit=5)
            
            for rel in related:
                expanded.append({
                    "source_entity": entity,
                    "related_entity": rel["name"],
                    "relation": rel["relation"],
                    "type": "graph_expansion"
                })
            
            # 获取前置知识
            prereqs = self.kg.find_prerequisites(entity, depth=2)
            for prereq in prereqs:
                expanded.append({
                    "source_entity": entity,
                    "prerequisite": prereq["name"],
                    "distance": prereq["distance"],
                    "type": "prerequisite"
                })
        
        return expanded[:top_k]
    
    def _hybrid_rank(self, vector_results: List[Dict], 
                    graph_results: List[Dict],
                    query: str) -> List[Dict]:
        """混合排序：结合向量相似度和图谱重要性"""
        combined_scores = {}
        
        # 向量结果打分
        for i, result in enumerate(vector_results):
            doc_id = result.get("id")
            # 向量相似度分数（归一化排名）
            vector_score = 1.0 - (i / len(vector_results)) if vector_results else 0
            combined_scores[doc_id] = {
                "document": result,
                "vector_score": vector_score,
                "graph_score": 0
            }
        
        # 图谱结果增强
        for graph_result in graph_results:
            # 找到对应的文档并加分
            for doc_id, scores in combined_scores.items():
                content = scores["document"].get("content", "")
                related_entity = graph_result.get("related_entity") or graph_result.get("prerequisite")
                
                if related_entity and related_entity in content:
                    scores["graph_score"] += 0.3  # 图谱匹配加分
        
        # 计算最终分数并排序
        final_results = []
        for doc_id, scores in combined_scores.items():
            final_score = scores["vector_score"] * 0.6 + scores["graph_score"] * 0.4
            result = scores["document"].copy()
            result["final_score"] = final_score
            result["vector_score"] = scores["vector_score"]
            result["graph_score"] = scores["graph_score"]
            final_results.append(result)
        
        # 按最终分数排序
        final_results.sort(key=lambda x: x["final_score"], reverse=True)
        
        return final_results
    
    def get_knowledge_context(self, query: str, concept_name: str) -> str:
        """获取知识点的完整上下文（用于 RAG 增强）"""
        context_parts = []
        
        # 1. 知识点基本信息
        concept = self.kg.get_concept(concept_name)
        if concept:
            context_parts.append(f"知识点：{concept['name']}")
            context_parts.append(f"定义：{concept.get('definition', '')}")
            context_parts.append(f"学科：{concept.get('subject', '')}")
            context_parts.append(f"年级：{concept.get('grade', '')}")
        
        # 2. 前置知识
        prereqs = self.kg.find_prerequisites(concept_name, depth=2)
        if prereqs:
            context_parts.append("\n前置知识：")
            for prereq in prereqs:
                context_parts.append(f"  - {prereq['name']}")
        
        # 3. 相关知识点
        related = self.kg.find_related(concept_name, limit=5)
        if related:
            context_parts.append("\n相关知识点：")
            for rel in related:
                context_parts.append(f"  - {rel['name']} ({rel['relation']})")
        
        return "\n".join(context_parts)
```

### 4.3 知识图谱构建脚本

```python
import json
from tqdm import tqdm

class KnowledgeGraphBuilder:
    """知识图谱构建器"""
    
    def __init__(self, kg: EducationKnowledgeGraph, llm_client):
        self.kg = kg
        self.llm = llm_client
    
    def build_from_textbook(self, textbook_data: List[Dict]):
        """从教材数据构建知识图谱"""
        print("开始构建知识图谱...")
        
        # 阶段 1: 提取实体
        print("\n[阶段 1/3] 提取知识点实体...")
        all_entities = []
        for chapter in tqdm(textbook_data):
            entities = self._extract_entities(chapter)
            all_entities.extend(entities)
        
        # 去重
        unique_entities = {e["name"]: e for e in all_entities}
        print(f"提取到 {len(unique_entities)} 个唯一实体")
        
        # 阶段 2: 创建实体节点
        print("\n[阶段 2/3] 创建实体节点...")
        for name, entity in tqdm(unique_entities.items()):
            self.kg.create_concept(
                name=name,
                subject=entity["subject"],
                grade=entity["grade"],
                definition=entity.get("definition", ""),
                properties=entity.get("properties", {})
            )
        
        # 阶段 3: 创建关系
        print("\n[阶段 3/3] 创建实体关系...")
        for chapter in tqdm(textbook_data):
            relationships = self._extract_relationships(chapter, unique_entities)
            for rel in relationships:
                self.kg.create_relationship(
                    from_name=rel["from"],
                    to_name=rel["to"],
                    rel_type=rel["type"],
                    properties=rel.get("properties", {})
                )
        
        print("\n知识图谱构建完成！")
    
    def _extract_entities(self, chapter: Dict) -> List[Dict]:
        """使用 LLM 从章节内容提取实体"""
        prompt = f"""
        从以下教学内容中提取知识点实体。
        
        章节：{chapter['title']}
        学科：{chapter['subject']}
        年级：{chapter['grade']}
        内容：
        {chapter['content'][:2000]}
        
        请以 JSON 格式输出实体列表：
        [
            {{
                "name": "实体名称",
                "type": "Concept",
                "definition": "简短定义"
            }}
        ]
        """
        
        response = self.llm.generate(prompt)
        try:
            entities = json.loads(response)
            # 添加学科和年级信息
            for e in entities:
                e["subject"] = chapter["subject"]
                e["grade"] = chapter["grade"]
            return entities
        except:
            return []
    
    def _extract_relationships(self, chapter: Dict, 
                               entities: Dict) -> List[Dict]:
        """使用 LLM 提取实体间关系"""
        entity_names = list(entities.keys())
        
        prompt = f"""
        分析以下知识点之间的关系。
        
        知识点列表：{entity_names}
        章节内容：{chapter['content'][:1500]}
        
        识别以下类型的关系：
        - PREREQUISITE_OF: 前置知识（必须先学A才能学B）
        - RELATED_TO: 相关知识点
        - REQUIRES: 需要/依赖（如：光合作用需要光照）
        - PART_OF: 组成部分
        
        请以 JSON 格式输出：
        [
            {{
                "from": "实体A",
                "to": "实体B",
                "type": "RELATIONSHIP_TYPE"
            }}
        ]
        """
        
        response = self.llm.generate(prompt)
        try:
            return json.loads(response)
        except:
            return []
```

## 五、面试要点

### 5.1 核心概念

1. **知识图谱 vs 向量数据库**：知识图谱擅长关系推理，向量数据库擅长语义匹配，两者互补
2. **GraphRAG 优势**：可解释性强、能发现隐含关系、支持推理
3. **Schema 设计原则**：实体类型清晰、关系定义明确、便于扩展

### 5.2 常见问题

**Q: 如何处理知识图谱的冷启动问题？**
A: ① 从结构化数据（教材目录、课程大纲）自动构建基础图谱 ② 使用 LLM 从非结构化文本抽取实体和关系 ③ 人工审核和补全

**Q: 知识图谱如何保证数据质量？**
A: ① 定义清晰的 Schema 约束 ② 实体消歧和链接 ③ 人工审核流程 ④ 持续更新和维护

**Q: GraphRAG 和传统 RAG 的区别？**
A: GraphRAG 在向量检索基础上增加了图谱推理，能发现实体间的隐含关系，返回更完整、结构化的答案。
