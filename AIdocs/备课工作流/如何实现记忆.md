# AI 教学对话助手的记忆机制实现

## 一、基础概念科普

### 1.1 为什么需要记忆机制？

在 AI 教学对话场景中，**记忆机制**是使助手能够"记住"用户信息、对话历史和上下文的关键技术：

| 场景 | 无记忆的问题 | 有记忆的优势 |
|------|-------------|-------------|
| **多轮对话** | 每次都要重复背景信息 | 自动理解上下文 |
| **个性化教学** | 无法针对学生特点调整 | 记住学生薄弱点重点讲解 |
| **长期辅导** | 每次对话都是新的开始 | 持续跟踪学习进度 |
| **知识关联** | 孤立回答每个问题 | 关联之前学过的知识点 |

## 四、代码实现

### 4.1 基础记忆类实现

```python
from typing import List, Dict, Optional, Any
from dataclasses import dataclass, field
from datetime import datetime
import json
import hashlib

@dataclass
class Message:
    """对话消息"""
    role: str  # 'human' 或 'ai'
    content: str
    timestamp: datetime = field(default_factory=datetime.now)
    metadata: Dict = field(default_factory=dict)
    
    def to_dict(self) -> Dict:
        return {
            "role": self.role,
            "content": self.content,
            "timestamp": self.timestamp.isoformat(),
            "metadata": self.metadata
        }

@dataclass
class ConversationSummary:
    """对话摘要"""
    summary: str
    key_points: List[str]
    topics: List[str]
    timestamp: datetime = field(default_factory=datetime.now)


class BaseMemory:
    """记忆基类"""
    
    def __init__(self):
        self.messages: List[Message] = []
    
    def add_message(self, role: str, content: str, metadata: Dict = None):
        """添加消息"""
        message = Message(role=role, content=content, metadata=metadata or {})
        self.messages.append(message)
    
    def get_memory(self) -> str:
        """获取记忆内容（子类实现）"""
        raise NotImplementedError
    
    def clear(self):
        """清空记忆"""
        self.messages = []


class ConversationBufferMemory(BaseMemory):
    """
    缓冲区记忆：保存所有历史对话
    适用于：短对话、需要完整上下文的场景
    """
    
    def __init__(self, max_token_limit: int = 4000):
        super().__init__()
        self.max_token_limit = max_token_limit
    
    def get_memory(self) -> str:
        """获取完整对话历史"""
        if not self.messages:
            return ""
        
        # 构建对话历史字符串
        history = []
        for msg in self.messages:
            role_label = "学生" if msg.role == "human" else "老师"
            history.append(f"{role_label}: {msg.content}")
        
        return "\n".join(history)
    
    def get_messages(self) -> List[Dict]:
        """获取消息列表（用于 API 调用）"""
        return [
            {"role": "user" if m.role == "human" else "assistant", 
             "content": m.content}
            for m in self.messages
        ]


class ConversationBufferWindowMemory(BaseMemory):
    """
    窗口记忆：只保留最近 k 轮对话
    适用于：控制上下文长度、长对话场景
    """
    
    def __init__(self, k: int = 5):
        super().__init__()
        self.k = k  # 保留的对话轮数
    
    def get_memory(self) -> str:
        """获取最近 k 轮对话"""
        # 只保留最后 2*k 条消息（k轮 = k条human + k条ai）
        recent_messages = self.messages[-2 * self.k:]
        
        history = []
        for msg in recent_messages:
            role_label = "学生" if msg.role == "human" else "老师"
            history.append(f"{role_label}: {msg.content}")
        
        return "\n".join(history)
    
    def get_messages(self) -> List[Dict]:
        """获取最近 k 轮消息"""
        recent_messages = self.messages[-2 * self.k:]
        return [
            {"role": "user" if m.role == "human" else "assistant", 
             "content": m.content}
            for m in recent_messages
        ]


class ConversationSummaryMemory(BaseMemory):
    """
    摘要记忆：使用 LLM 生成对话摘要
    适用于：超长对话、需要节省 Token 的场景
    """
    
    def __init__(self, llm_client, summary_interval: int = 6):
        super().__init__()
        self.llm = llm_client
        self.summary_interval = summary_interval  # 每隔多少条消息生成摘要
        self.summary: Optional[ConversationSummary] = None
        self.buffer: List[Message] = []  # 未摘要的缓冲消息
    
    def add_message(self, role: str, content: str, metadata: Dict = None):
        """添加消息，触发摘要生成"""
        super().add_message(role, content, metadata)
        self.buffer.append(self.messages[-1])
        
        # 达到阈值时生成摘要
        if len(self.buffer) >= self.summary_interval:
            self._generate_summary()
    
    def _generate_summary(self):
        """生成对话摘要"""
        # 构建摘要生成 Prompt
        conversation_text = "\n".join([
            f"{'学生' if m.role == 'human' else '老师'}: {m.content}"
            for m in self.buffer
        ])
        
        prompt = f"""请对以下教学对话进行摘要，提取关键信息：

对话内容：
{conversation_text}

请以 JSON 格式输出：
{{
    "summary": "整体摘要（50字以内）",
    "key_points": ["要点1", "要点2"],
    "topics": ["涉及的知识点1", "知识点2"],
    "student_status": "学生的学习状态（掌握/困惑/进步）"
}}"""
        
        # 调用 LLM 生成摘要
        response = self.llm.generate(prompt)
        try:
            summary_data = json.loads(response)
            self.summary = ConversationSummary(
                summary=summary_data.get("summary", ""),
                key_points=summary_data.get("key_points", []),
                topics=summary_data.get("topics", [])
            )
            # 清空缓冲
            self.buffer = []
        except:
            pass
    
    def get_memory(self) -> str:
        """获取摘要 + 缓冲消息"""
        parts = []
        
        # 添加历史摘要
        if self.summary:
            parts.append(f"【历史对话摘要】{self.summary.summary}")
            if self.summary.key_points:
                parts.append(f"关键要点：{', '.join(self.summary.key_points)}")
        
        # 添加缓冲消息
        if self.buffer:
            parts.append("【最近对话】")
            for msg in self.buffer:
                role_label = "学生" if msg.role == "human" else "老师"
                parts.append(f"{role_label}: {msg.content}")
        
        return "\n".join(parts)
```

### 4.2 长期记忆与用户画像

```python
import redis
import psycopg2
from typing import Optional

class LongTermMemory:
    """
    长期记忆管理
    存储：用户画像、学习历史、知识掌握度
    """
    
    def __init__(self, redis_host='localhost', pg_config=None):
        self.redis = redis.Redis(host=redis_host, port=6379, db=0)
        self.pg_config = pg_config or {
            'host': 'localhost',
            'database': 'education_db',
            'user': 'postgres',
            'password': 'password'
        }
    
    def save_user_profile(self, user_id: str, profile: Dict):
        """保存用户画像"""
        key = f"user:{user_id}:profile"
        self.redis.setex(key, 86400 * 30, json.dumps(profile))  # 30天过期
    
    def get_user_profile(self, user_id: str) -> Optional[Dict]:
        """获取用户画像"""
        key = f"user:{user_id}:profile"
        data = self.redis.get(key)
        return json.loads(data) if data else None
    
    def record_learning_progress(self, user_id: str, concept: str, 
                                 status: str, timestamp: datetime = None):
        """记录学习进度"""
        if timestamp is None:
            timestamp = datetime.now()
        
        # 存入 PostgreSQL
        conn = psycopg2.connect(**self.pg_config)
        cur = conn.cursor()
        
        cur.execute("""
            INSERT INTO learning_progress 
            (user_id, concept, status, timestamp)
            VALUES (%s, %s, %s, %s)
            ON CONFLICT (user_id, concept) 
            DO UPDATE SET status = EXCLUDED.status, timestamp = EXCLUDED.timestamp
        """, (user_id, concept, status, timestamp))
        
        conn.commit()
        cur.close()
        conn.close()
    
    def get_weak_points(self, user_id: str, subject: str = None, 
                       limit: int = 10) -> List[Dict]:
        """获取用户的薄弱知识点"""
        conn = psycopg2.connect(**self.pg_config)
        cur = conn.cursor()
        
        if subject:
            cur.execute("""
                SELECT concept, status, timestamp
                FROM learning_progress
                WHERE user_id = %s AND status IN ('confused', 'not_mastered')
                AND concept IN (SELECT name FROM concepts WHERE subject = %s)
                ORDER BY timestamp DESC
                LIMIT %s
            """, (user_id, subject, limit))
        else:
            cur.execute("""
                SELECT concept, status, timestamp
                FROM learning_progress
                WHERE user_id = %s AND status IN ('confused', 'not_mastered')
                ORDER BY timestamp DESC
                LIMIT %s
            """, (user_id, limit))
        
        results = cur.fetchall()
        cur.close()
        conn.close()
        
        return [
            {"concept": r[0], "status": r[1], "timestamp": r[2]}
            for r in results
        ]
    
    def update_concept_mastery(self, user_id: str, concept: str, 
                               score: float, interaction_count: int):
        """更新知识点掌握度"""
        key = f"user:{user_id}:mastery:{concept}"
        data = {
            "concept": concept,
            "score": score,  # 0-1 掌握度分数
            "interaction_count": interaction_count,
            "last_updated": datetime.now().isoformat()
        }
        self.redis.setex(key, 86400 * 90, json.dumps(data))  # 90天过期
    
    def get_concept_mastery(self, user_id: str, concept: str) -> Optional[Dict]:
        """获取知识点掌握度"""
        key = f"user:{user_id}:mastery:{concept}"
        data = self.redis.get(key)
        return json.loads(data) if data else None


class UserProfile:
    """用户画像数据模型"""
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.grade: str = ""  # 年级
        self.subjects: List[str] = []  # 关注学科
        self.learning_style: str = ""  # 学习风格（visual/auditory/kinesthetic）
        self.preferred_difficulty: str = "medium"  # 偏好难度
        self.common_mistakes: List[str] = []  # 常见错误类型
        self.interests: List[str] = []  # 兴趣领域
        self.total_study_time: int = 0  # 总学习时长（分钟）
        self.created_at: datetime = datetime.now()
        self.updated_at: datetime = datetime.now()
    
    def to_dict(self) -> Dict:
        return {
            "user_id": self.user_id,
            "grade": self.grade,
            "subjects": self.subjects,
            "learning_style": self.learning_style,
            "preferred_difficulty": self.preferred_difficulty,
            "common_mistakes": self.common_mistakes,
            "interests": self.interests,
            "total_study_time": self.total_study_time,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat()
        }
    
    @classmethod
    def from_dict(cls, data: Dict) -> 'UserProfile':
        profile = cls(data["user_id"])
        profile.grade = data.get("grade", "")
        profile.subjects = data.get("subjects", [])
        profile.learning_style = data.get("learning_style", "")
        profile.preferred_difficulty = data.get("preferred_difficulty", "medium")
        profile.common_mistakes = data.get("common_mistakes", [])
        profile.interests = data.get("interests", [])
        profile.total_study_time = data.get("total_study_time", 0)
        return profile
```

### 4.3 智能记忆管理器

```python
class SmartMemoryManager:
    """
    智能记忆管理器
    整合多种记忆类型，自动选择最佳策略
    """
    
    def __init__(self, llm_client, vector_store, long_term_memory):
        self.llm = llm_client
        self.vector_store = vector_store
        self.ltm = long_term_memory
        
        # 初始化不同记忆类型
        self.buffer_memory = ConversationBufferMemory()
        self.window_memory = ConversationBufferWindowMemory(k=5)
        self.summary_memory = ConversationSummaryMemory(llm_client)
        
        self.current_memory_type = "window"  # 默认使用窗口记忆
    
    def process_message(self, user_id: str, message: str, 
                       context: Dict = None) -> str:
        """
        处理用户消息，整合记忆并生成回复
        
        流程：
        1. 检索长期记忆（用户画像、薄弱点）
        2. 获取短期记忆（对话历史）
        3. 向量检索相关知识
        4. 构建增强提示
        5. 生成回复
        6. 更新记忆
        """
        context = context or {}
        
        # 1. 检索长期记忆
        user_profile = self.ltm.get_user_profile(user_id)
        weak_points = self.ltm.get_weak_points(user_id, context.get("subject"))
        
        # 2. 获取短期记忆
        short_term = self._get_short_term_memory()
        
        # 3. 向量检索
        vector_results = self.vector_store.search(message, top_k=3)
        
        # 4. 构建增强提示
        system_prompt = self._build_system_prompt(user_profile, weak_points)
        memory_context = self._build_memory_context(short_term, vector_results)
        
        # 5. 生成回复
        messages = [
            {"role": "system", "content": system_prompt},
            {"role": "system", "content": f"相关背景知识：\n{memory_context}"},
        ]
        
        # 添加历史对话
        if self.current_memory_type == "buffer":
            messages.extend(self.buffer_memory.get_messages())
        elif self.current_memory_type == "window":
            messages.extend(self.window_memory.get_messages())
        elif self.current_memory_type == "summary":
            messages.append({
                "role": "system", 
                "content": self.summary_memory.get_memory()
            })
        
        # 添加当前消息
        messages.append({"role": "user", "content": message})
        
        # 调用 LLM
        response = self.llm.generate(messages)
        
        # 6. 更新记忆
        self._update_memories(user_id, message, response, context)
        
        return response
    
    def _get_short_term_memory(self) -> str:
        """获取短期记忆内容"""
        if self.current_memory_type == "buffer":
            return self.buffer_memory.get_memory()
        elif self.current_memory_type == "window":
            return self.window_memory.get_memory()
        elif self.current_memory_type == "summary":
            return self.summary_memory.get_memory()
        return ""
    
    def _build_system_prompt(self, user_profile: Optional[Dict], 
                            weak_points: List[Dict]) -> str:
        """构建系统提示"""
        parts = ["你是一位专业的教学助手，擅长个性化辅导。"]
        
        if user_profile:
            parts.append(f"学生年级：{user_profile.get('grade', '未知')}")
            parts.append(f"学习风格：{user_profile.get('learning_style', '未设定')}")
            parts.append(f"偏好难度：{user_profile.get('preferred_difficulty', '中等')}")
        
        if weak_points:
            concepts = [wp["concept"] for wp in weak_points[:3]]
            parts.append(f"学生近期薄弱知识点：{', '.join(concepts)}")
            parts.append("请在回答中重点关注这些薄弱点，给予更多解释和练习。")
        
        return "\n".join(parts)
    
    def _build_memory_context(self, short_term: str, 
                             vector_results: List[Dict]) -> str:
        """构建记忆上下文"""
        parts = []
        
        if short_term:
            parts.append(f"【对话历史】\n{short_term}")
        
        if vector_results:
            parts.append("【相关知识】")
            for result in vector_results:
                parts.append(f"- {result['content'][:200]}")
        
        return "\n".join(parts)
    
    def _update_memories(self, user_id: str, message: str, 
                        response: str, context: Dict):
        """更新各类记忆"""
        # 更新短期记忆
        self.buffer_memory.add_message("human", message, context)
        self.buffer_memory.add_message("ai", response)
        
        self.window_memory.add_message("human", message, context)
        self.window_memory.add_message("ai", response)
        
        self.summary_memory.add_message("human", message, context)
        self.summary_memory.add_message("ai", response)
        
        # 更新长期记忆（异步）
        # 分析对话内容，提取学习状态
        if "不懂" in message or "不明白" in message:
            concept = context.get("current_concept", "unknown")
            self.ltm.record_learning_progress(user_id, concept, "confused")
        elif "懂了" in message or "明白了" in message:
            concept = context.get("current_concept", "unknown")
            self.ltm.record_learning_progress(user_id, concept, "mastered")
    
    def switch_memory_type(self, memory_type: str):
        """切换记忆类型"""
        if memory_type in ["buffer", "window", "summary"]:
            self.current_memory_type = memory_type
            print(f"已切换到 {memory_type} 记忆模式")
        else:
            raise ValueError(f"未知的记忆类型: {memory_type}")
```

## 五、面试要点

### 5.1 核心概念

1. **短期记忆 vs 长期记忆**：短期记忆用于当前对话上下文，长期记忆用于跨会话的用户画像和学习历史
2. **记忆类型选择**：根据对话长度和场景选择合适的记忆策略（Buffer/Window/Summary）
3. **记忆更新策略**：何时更新、如何提取关键信息、如何处理冲突

### 5.2 常见问题

**Q: 如何处理超长对话的上下文限制？**
A: ① 使用窗口记忆限制历史长度 ② 使用摘要记忆压缩历史 ③ 结合向量检索补充相关历史

**Q: 如何保证长期记忆的准确性？**
A: ① 多源验证（对话内容、测试结果、学习行为）② 定期清理过期记忆 ③ 用户反馈机制

**Q: 记忆机制如何与 RAG 结合？**
A: 短期记忆提供对话上下文，长期记忆提供用户画像，向量检索提供知识增强，三者共同构建完整的上下文
